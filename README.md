기본 구조 예시

your-repo-name/
├── index.html            # 메인 페이지 (전체 학습 내용의 목차)
│
├── css/                  # CSS 스타일시트 폴더
│   └── style.css         # 사이트 전체에 적용될 스타일
│
├── images/               # 이미지 파일 폴더
│   └── profile.png       # 프로필 사진 등
│
└── posts/                # 학습 내용을 저장할 폴더 (가장 중요!)
    │
    ├── html/             # HTML 관련 학습 내용
    │   ├── 2023-10-27-html-basics.html
    │   └── 2023-10-28-semantic-tags.html
    │
    ├── css/              # CSS 관련 학습 내용
    │   └── 2023-11-01-flexbox-guide.html
    │
    └── javascript/       # JavaScript 관련 학습 내용
        └── 2023-11-05-what-is-dom.html
